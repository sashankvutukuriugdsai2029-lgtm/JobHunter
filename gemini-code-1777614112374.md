## LangGraph Global State (TypedDict)
```python
from typing import TypedDict, List, Optional
from pydantic import BaseModel, Field

class CandidateProfile(BaseModel):
    name: str
    base_skills: List[str]
    years_experience: int
    visa_status: str = Field(description="e.g., Citizen, Requires Sponsorship, H1B")
    location: str = Field(description="City/State/Country")

class SearchStrategy(BaseModel):
    target_seniority: str = Field(description="e.g., Entry-Level, Mid-Level, Senior")
    target_role_titles: List[str] = Field(description="e.g., Frontend Developer, Fullstack Engineer")
    target_salary: int = Field(description="Minimum expected salary/stipend")
    work_model: str = Field(description="Remote, Hybrid, or On-site")
    dealbreakers: List[str] = Field(description="e.g., No crypto, Must have health insurance")
    prep_recommendations: List[str] = Field(description="Resources the user should study based on feedback")

class Application(BaseModel):
    job_id: str
    company_name: str
    role_title: str
    status: str = Field(description="Must be: applied, screening, interviewing, rejected, offer")
    date_applied: str

class FeedbackEvent(BaseModel):
    application_id: str
    raw_email_text: str
    extracted_reason: str
    date_logged: str

class Pattern(BaseModel):
    theme: str = Field(description="The recurring issue, e.g., 'System Design gap'")
    occurrences: int = Field(description="Number of times this theme appeared")
    proposed_action: str = Field(description="What the agent wants to do about it")

class JobHunterState(TypedDict):
    candidate_profile: Optional[CandidateProfile]
    current_strategy: Optional[SearchStrategy]
    applications: List[Application]
    feedback_log: List[FeedbackEvent]
    rejection_patterns: List[Pattern]
    is_onboarded: bool
