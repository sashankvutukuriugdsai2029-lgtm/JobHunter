from pydantic import BaseModel
from typing import List, Optional
from enum import Enum

class ApplicationStatus(str, Enum):
    APPLIED = "applied"
    SCREENING = "screening"
    INTERVIEWING = "interviewing"
    REJECTED = "rejected"
    OFFER = "offer"

class Application(BaseModel):
    job_id: str
    company: str
    role: str
    status: ApplicationStatus
    applied_date: str

class FeedbackEvent(BaseModel):
    job_id: str
    feedback_text: str
    date: str

class Pattern(BaseModel):
    theme: str          # e.g., "system design weakness"
    frequency: int      # how many times detected
    suggested_action: str

class SearchStrategy(BaseModel):
    seniority_level: str    # junior / mid / senior
    role_types: List[str]   # ["SDE", "Backend Engineer"]
    prep_resources: List[str]

class CandidateProfile(BaseModel):
    name: str
    skills: List[str]
    target_roles: List[str]
    experience_years: int

    sessions_count: int = 0

Node 7: pattern_detector.py (Most Important Node)
python
# Look at feedback_log
# If 3+ feedbacks mention similar keywords → create a Pattern
# Keywords: "system design", "communication", "coding", "culture fit"
def pattern_detector(state: JobHunterState) -> JobHunterState:
    if len(state.feedback_log) < 3:
        return state   # not enough data yet
    
    # Count keyword occurrences across all feedback
    keyword_counts = {}
    for event in state.feedback_log:
        for keyword in ["system design", "communication", "coding"]:
            if keyword in event.feedback_text.lower():
                keyword_counts[keyword] = keyword_counts.get(keyword, 0) + 1
    
    # If any keyword appears 3+ times, it's a pattern
    for keyword, count in keyword_counts.items():
        if count >= 3:
            pattern = Pattern(
                theme=keyword,
                frequency=count,
                suggested_action=f"Practice {keyword} — see resources"
            )
            state.rejection_patterns.append(pattern)
    
    return state

from langgraph.graph import StateGraph
from langgraph.checkpoint.sqlite import SqliteSaver
from state import JobHunterState
from nodes import *

# Create the graph
builder = StateGraph(JobHunterState)

# Add all nodes
builder.add_node("state_loader", state_loader)
builder.add_node("job_fetcher", job_fetcher)
builder.add_node("matcher", matcher)
builder.add_node("ranker", ranker)
builder.add_node("application_tracker", application_tracker)
builder.add_node("feedback_parser", feedback_parser)
builder.add_node("pattern_detector", pattern_detector)
builder.add_node("strategy_updater", strategy_updater)
builder.add_node("state_saver", state_saver)

# Define the flow
builder.set_entry_point("state_loader")
builder.add_edge("state_loader", "job_fetcher")
builder.add_edge("job_fetcher", "matcher")
builder.add_edge("matcher", "ranker")
builder.add_edge("ranker", "application_tracker")
builder.add_edge("application_tracker", "feedback_parser")
builder.add_edge("feedback_parser", "pattern_detector")
builder.add_edge("pattern_detector", "strategy_updater")
builder.add_edge("strategy_updater", "state_saver")

# Add SQLite checkpointer (this is what makes it stateful!)
checkpointer = SqliteSaver.from_conn_string("jobhunter.db")
graph = builder.compile(checkpointer=checkpointer)

1. Company Background
Active job seekers apply to 50 to 100 roles over a month. They manage applications in spreadsheets, tailor resumes manually, and forget which feedback applies to which role. Today's job boards show listings but do not learn from a candidate's evolving outcomes. 
Pathmark wants an agent that maintains the candidate's full search state — profile, applications, outcomes — and adapts its recommendations as feedback comes in. 
2. The Business Problem
Build a stateful agent that, on each session, fetches new matching jobs, ranks them against the candidate's evolving profile, tracks new applications the candidate logs, parses feedback from rejections and offers, and updates the search strategy. 
The agent must detect patterns in rejections (e.g., 'failing at system design rounds') and surface specific preparation resources or adjust the seniority of the roles it recommends. 
3. Why LangGraph (Required Pattern)
This problem exercises persistent state, pattern detection over historical data, and strategy adaptation — all requiring more than a single prompt. LangGraph's checkpointer plus conditional logic around the historical state is the right abstraction. 
4. Required Graph Architecture
4.1. Minimum nodes (build all of these):

state_loader — Loads candidate profile, application history, feedback log
job_fetcher — Fetches new roles matching the current strategy
matcher — Scores each role against profile and learned preferences
ranker — Applies evolving fit-score logic, returns top 10
application_tracker — Records new applications logged by the candidate
feedback_parser — Processes candidate-entered rejection/offer / interview feedback
pattern_detector — Detects recurring themes in rejections
strategy_updater — Adjusts seniority level, role type, or prep recommendations based on patterns
state_saver — Checkpoints everything for next session   
4.2. Graph State (Pydantic BaseModel fields):

candidate_profile: CandidateProfile
applications: list[Application] (with status: applied, rejected, interviewing, offer)
feedback_log: list[FeedbackEvent]
rejection_patterns: list[Pattern]
current_strategy: SearchStrategy
sessions_count: int 4.3. Pattern-specific requirement: Persistent checkpointer required. Pattern detection must examine at least 3 historical events before triggering a strategy change. Strategy changes must be explicit and explainable in the UI.

Class 1: ApplicationStatus (Enum)
What is an Enum?
An Enum is just a fixed list of allowed values. Like a dropdown menu — you can only pick from what's listed.

Why do we need it?
Without enum, someone could type status = "appliedd" (typo) and the app breaks silently. With enum, only valid values are allowed.

python
# Without enum — dangerous
status = "appliedd"   # typo, no error raised, silent bug

# With enum — safe
status = ApplicationStatus.APPLIED   # if wrong, immediate error

python
import pandas as pd

# Load the full dataset
df = pd.read_csv("data/raw_jobs.csv")
print(f"Original dataset: {len(df)} rows, {len(df.columns)} columns")

# Keep only the columns we need
columns_to_keep = [
    "job_id",
    "company_name",
    "title",
    "location",
    "skills_desc",
    "normalized_salary",
    "formatted_work_type"      # keeping this for reference
]

df_clean = df[columns_to_keep].copy()

# Clean step 1: Drop rows where title or company is missing
df_clean = df_clean.dropna(subset=["title", "company_name"])

# Clean step 2: Fill missing values
df_clean["location"] = df_clean["location"].fillna("Remote")
df_clean["skills_desc"] = df_clean["skills_desc"].fillna("")
df_clean["normalized_salary"] = df_clean["normalized_salary"].fillna(0)
df_clean["formatted_work_type"] = df_clean["formatted_work_type"].fillna("Full-time")

# Clean step 3: Derive seniority_level from job title
def derive_seniority(title):
    title_lower = str(title).lower()
    if any(word in title_lower for word in ["intern", "internship", "trainee"]):
        return "internship"
    elif any(word in title_lower for word in ["junior", "jr", "entry", "associate", "graduate"]):
        return "junior"
    elif any(word in title_lower for word in ["senior", "sr", "lead", "principal", "staff"]):
        return "senior"
    elif any(word in title_lower for word in ["director", "vp", "head", "manager", "chief"]):
        return "director"
    else:
        return "mid"       # default — no keyword found

df_clean["seniority_level"] = df_clean["title"].apply(derive_seniority)

# Clean step 4: Keep only tech related jobs
tech_keywords = [
    "engineer", "developer", "software", "backend",
    "frontend", "fullstack", "full stack", "data",
    "python", "java", "react", "node", "devops",
    "cloud", "ml", "machine learning", "ai", "analyst"
]
mask = df_clean["title"].str.lower().str.contains(
    "|".join(tech_keywords),
    na=False
)
df_clean = df_clean[mask]

# Clean step 5: Keep only Full-time and Internship work types
df_clean = df_clean[df_clean["formatted_work_type"].isin(["Full-time", "Internship", "Contract"])]

# Clean step 6: Take 500 rows
df_clean = df_clean.head(500).reset_index(drop=True)

# Save
df_clean.to_csv("data/jobs.csv", index=False)

print(f"\nCleaned dataset: {len(df_clean)} rows")
print(f"Columns: {list(df_clean.columns)}")
print(f"\nSeniority levels derived from titles:")
print(df_clean["seniority_level"].value_counts())
print(f"\nWork types:")
print(df_clean["formatted_work_type"].value_counts())
print(f"\nSample rows:")
print(df_clean[["job_id","company_name","title","seniority_level","location"]].head(5).to_string())
print(f"\nSaved to data/jobs.csv ✅")

def job_fetcher(state: JobHunterState) -> dict:

    Second node in the LangGraph workflow.
    Reads jobs.csv and filters them based on the candidate's

    Returns matching jobs as a list of dicts in recommended_jobs.
    Empty locations list = no location filter (show all locations).
    Role types are split into keywords for broader matching.

        print("[job_fetcher] No current strategy found. Skipped.")
        return {
            "recommended_jobs": [],
            "session_summary": "No strategy set. Cannot fetch jobs."
        }

    seniority = state.current_strategy.seniority_level
    locations = state.current_strategy.locations
    role_types = state.current_strategy.role_types
    avoid_keywords = state.current_strategy.avoid_keywords

    print(f"[job_fetcher] Seniority: {seniority}")
    print(f"[job_fetcher] Locations: {locations}")
    print(f"[job_fetcher] Role types: {role_types}")

    BASE_DIR = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
    csv_path = os.path.join(BASE_DIR, "data", "jobs.csv")

    try:
        df = pd.read_csv(csv_path)
        print(f"[job_fetcher] Total jobs in CSV: {len(df)}")
    except Exception as e:
        print(f"[job_fetcher] Error loading jobs data: {e}")
        return {
            "recommended_jobs": [],
            "session_summary": "Error loading jobs database."
        }

    # STEP 1: Map seniority to experience levels
    experience_map = {
        "junior": ["Entry level", "Associate", "Internship"],
        "mid":    ["Mid-Senior level", "Associate"],
        "senior": ["Mid-Senior level", "Director"]
    }
    target_experience = experience_map.get(
        seniority.lower(), experience_map["junior"])
    df = df[df["formatted_experience_level"].isin(target_experience)]
    print(f"[job_fetcher] After seniority filter: {len(df)} jobs")

    # STEP 2: Filter by location
    # Empty locations list = no filter, show all locations
    if locations:

        is_remote = df["remote_allowed"].astype(str).str.lower().isin(
            ["true", "1", "1.0"])
        df = df[contain_loc | is_remote]
        print(f"[job_fetcher] After location filter: {len(df)} jobs")
    else:
        print(f"[job_fetcher] No location filter — showing all locations")

    # STEP 3: Filter by role types using keyword matching
    if role_types:
        role_keywords = []
        for role in role_types:
            words = role.lower().split()
            role_keywords.extend(words)
        stop_words = ["and", "or", "the", "a", "an", "of", "in"]
        role_keywords = [w for w in role_keywords if w not in stop_words]
        role_keywords = list(set(role_keywords))
        role_pattern = "|".join(role_keywords)
        df = df[df["title"].str.lower().str.contains(
            role_pattern, na=False)]
        print(f"[job_fetcher] After role filter: {len(df)} jobs")
    else:
        print(f"[job_fetcher] No role filter — showing all roles")

    # STEP 4: Remove avoid_keywords
    if avoid_keywords:
        avoid_pattern = "|".join(
            [str(kw).lower() for kw in avoid_keywords])
        mask1 = df["title"].str.lower().str.contains(
            avoid_pattern, na=False)
        mask2 = df["description"].str.lower().str.contains(
            avoid_pattern, na=False)
        df = df[~(mask1 | mask2)]
        print(f"[job_fetcher] After avoid filter: {len(df)} jobs")

    print(f"[job_fetcher] Final jobs found: {len(df)}")

    jobs_list = df.to_dict(orient="records")

    return {
        "recommended_jobs": jobs_list,
        "session_summary": f"Found {len(jobs_list)} matching jobs"
    }

def calculate_match_score(job: dict, 
                          profile: CandidateProfile) -> int:
    score = 0

    SKILLS CHECK (max 40 points):
    - Get candidate skills list from profile.skills
    - For each skill in candidate skills:
        check if skill appears in job["description"]
        OR job["skills_desc"] (case insensitive)
    - points_per_skill = 40 / len(skills) if skills else 0
    - Add points for each matching skill
    - Cap at 40 points maximum

    ROLE MATCH CHECK (max 30 points):
    - Get candidate target_roles from profile.target_roles
    - Check if any target_role word appears in job["title"]
      (case insensitive, split into keywords)
    - If match found → add 30 points

    EXPERIENCE LEVEL CHECK (max 20 points):
    - Map candidate experience_years to expected level:
        0-1 years  → "Entry level" or "Internship"
        2-4 years  → "Associate" or "Mid-Senior level"

    - If job formatted_experience_level matches 
      expected level → add 20 points

    REMOTE BONUS (max 10 points):
    - If job remote_allowed is True → add 10 points

    return min(score, 100)  # cap at 100

    """

        )

        base_url=base_url
    )
    return client

def get_model_name() -> str:
    """Returns the model name from .env"""
    return os.getenv(
        "OPENROUTER_MODEL", 
        "google/gemma-3-27b-it:free"
    )

def call_llm(prompt: str, 
             system: str = "You are a helpful assistant.",
             max_tokens: int = 500) -> str:
    """
    Simple wrapper to call the LLM.
    Takes a prompt string, returns response string.
    All LLM nodes use this function.
    """
    client = get_llm_client()
    model = get_model_name()
    
    try:
        response = client.chat.completions.create(
            model=model,
            messages=[
                {"role": "system", "content": system},
                {"role": "user", "content": prompt}
            ],
            max_tokens=max_tokens
        )
        return response.choices[0].message.content.strip()
    except Exception as e:
        print(f"[LLM ERROR] {e}")
        return ""

pythondef call_llm(prompt: str, 
             system: str = "You are a helpful assistant.",
             max_tokens: int = 500) -> str:
    client = get_llm_client()
    model = get_model_name()
    
    try:
        response = client.chat.completions.create(
            model=model,
            messages=[
                {"role": "system", "content": system},
                {"role": "user", "content": prompt}
            ],
            max_tokens=max_tokens
        )
        message = response.choices[0].message
        
        # Try content first (normal models)
        if message.content and message.content.strip():
            return message.content.strip()
        
        # Reasoning models put answer in reasoning field
        # Extract just the final answer from reasoning text
        if hasattr(message, 'reasoning') and message.reasoning:
            reasoning_text = message.reasoning
            # The reasoning model thinks out loud then concludes
            # Take the last sentence which is usually the answer
            lines = [l.strip() for l in reasoning_text.split('\n') 
                     if l.strip()]
            # Look for a line that looks like keywords (has commas)
            for line in reversed(lines):
                if ',' in line and len(line) < 200:
                    return line.strip()
            # fallback: return last non-empty line
            if lines:
                return lines[-1]
        
        return ""
    except Exception as e:
        print(f"[LLM ERROR] {type(e).__name__}: {e}")
        return ""
